toggle_marrying_off_unworthy_courtiers = {
	scope = character
	is_shown = {
		has_global_variable = disable_marrying_off_unworthy_courtiers
	}
	effect = {
		if = {
			limit = {
				has_global_variable = disable_marrying_off_unworthy_courtiers
			}
			remove_global_variable = disable_marrying_off_unworthy_courtiers
		}
		else = {
			set_global_variable = disable_marrying_off_unworthy_courtiers
		}
	}
}
